<objective>
Implement the core hibernation engine: the `handleAlarmTick()` function that fires every 1 minute via the `HIBERNATE_CHECK` alarm. It reads the global enabled flag and all tab metadata from chrome.storage.local, queries every open tab, applies the `isDiscardable()` guard against all protection conditions (active, audible, pinned, discarded, loading, protected tab ID, protected domain, form activity within 5 min, inactivity less than 45 min), calls `chrome.tabs.discard()` on eligible tabs, increments `hibernated_count`, and updates the icon badge via `updateBadge()`.

This plan also ensures `handleManualHibernate()` updates the count and badge after a successful manual discard.

After this plan, the extension hibernates tabs automatically and all badge/count state is correct.
</objective>

<must_haves>
- `handleAlarmTick()` reads `hibernation_enabled` from storage and returns early without discarding if false
- `handleAlarmTick()` calls `chrome.tabs.query({})` to get all open tabs (not filtered by URL — `isDiscardable` handles URL filtering)
- `handleAlarmTick()` calls `isDiscardable(tab, meta, now, protected_tabs, protected_domains)` for every tab
- `handleAlarmTick()` calls `chrome.tabs.discard(tab.id)` for each eligible tab and awaits the result
- `handleAlarmTick()` increments `hibernated_count` in storage for each successfully discarded tab (discard returns a Tab object, not undefined)
- `handleAlarmTick()` calls `updateBadge(count)` after all discards complete
- `handleManualHibernate()` calls `updateBadge()` after successful discard
- `src/background/hibernation.ts` imports `updateBadge` from `./badge` and `getStorage`/`setStorage` from `../shared/types`
</must_haves>

<threat_model>
| Threat | STRIDE | Mitigation |
|--------|--------|------------|
| Global in-memory state lost on SW restart | Tampering | All state read fresh from chrome.storage.local on every alarm tick — no module-level variables hold tab state |
| Concurrent alarm ticks race on storage | Tampering | alarm period is 1 minute; Chrome does not fire alarms faster than 30s minimum; wrap handler in try/finally to prevent stuck state |
| Discarding a tab that becomes active between query and discard | Denial of Service | isDiscardable() guards tab.active; chrome.tabs.discard() no-ops on active tabs (returns undefined); undefined return is handled gracefully |
</threat_model>

import type { TabMeta } from '../shared/types'
import { TIMEOUT_MS, FORM_PROTECTION_MS } from '../shared/constants'
import { updateBadge } from './badge'

export function isDiscardable(
  tab: chrome.tabs.Tab,
  meta: TabMeta | undefined,
  now: number,
  protectedTabs: number[],
  protectedDomains: string[]
): boolean {
  if (!tab.id || !tab.url) return false
  if (!tab.url.startsWith('http://') && !tab.url.startsWith('https://')) return false
  if (tab.active) return false
  if (tab.audible) return false
  if (tab.pinned) return false
  if (tab.discarded) return false
  if (tab.status !== 'complete') return false
  if (protectedTabs.includes(tab.id)) return false
  try {
    const hostname = new URL(tab.url).hostname
    if (protectedDomains.includes(hostname)) return false
  } catch {
    return false
  }
  const lastActive = meta?.lastActiveAt ?? now
  if (now - lastActive < TIMEOUT_MS) return false
  if (meta?.lastFormActivity && now - meta.lastFormActivity < FORM_PROTECTION_MS) return false
  return true
}

export async function handleAlarmTick(): Promise<void> {
  // Read all required state in a single storage.get call for atomicity
  const result = await chrome.storage.local.get([
    'hibernation_enabled',
    'tab_meta',
    'protected_tabs',
    'protected_domains',
    'hibernated_count',
  ])

  const hibernationEnabled = result['hibernation_enabled'] as boolean ?? true
  if (!hibernationEnabled) return

  const tabMeta = (result['tab_meta'] as Record<number, TabMeta>) ?? {}
  const protectedTabs = (result['protected_tabs'] as number[]) ?? []
  const protectedDomains = (result['protected_domains'] as string[]) ?? []
  let hibernatedCount = (result['hibernated_count'] as number) ?? 0

  // Query ALL tabs — isDiscardable handles URL/state filtering
  const tabs = await chrome.tabs.query({})
  const now = Date.now()

  let newDiscards = 0

  for (const tab of tabs) {
    if (!tab.id) continue
    const meta = tabMeta[tab.id]
    if (!isDiscardable(tab, meta, now, protectedTabs, protectedDomains)) continue

    try {
      const discarded = await chrome.tabs.discard(tab.id)
      // discard() returns undefined when the tab cannot be discarded (already discarded, active, protected)
      if (discarded !== undefined) {
        newDiscards++
      }
    } catch {
      // Tab may have been closed between query and discard — silently continue
    }
  }

  if (newDiscards > 0) {
    hibernatedCount += newDiscards
    await chrome.storage.local.set({ hibernated_count: hibernatedCount })
    await updateBadge(hibernatedCount)
  }
}

export async function handleManualHibernate(tabId: number): Promise<void> {
  try {
    const discarded = await chrome.tabs.discard(tabId)
    if (discarded !== undefined) {
      const result = await chrome.storage.local.get('hibernated_count')
      const count = ((result['hibernated_count'] as number) ?? 0) + 1
      await chrome.storage.local.set({ hibernated_count: count })
      await updateBadge(count)
    }
  } catch {
    // Tab closed or not discardable — ignore
  }
}
